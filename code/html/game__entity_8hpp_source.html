384  void updateNode() {
385  // End early if not alive
386  if (!alive) return;
387 
388  // First, update itself
389  update();
390 
391  // End early if not alive (again after update)
392  if (!alive) return;
393 
394  // If there is any update priority change among childs
395  if (std::any_of(childNodes.begin(), childNodes.end(), [](GameEntity *e){return e->priorityChanged();})) {
396  // Re-sort vector
397  std::sort(childNodes.begin(), childNodes.end(), comparePtrToObj);
398  // Set all as clean
399  std::for_each(childNodes.begin(), childNodes.end(), [](GameEntity *e){e->acknowledgePriorityChange();});
400  }
401 
402  // Update childs (work on a copy or GG)
403  std::vector<GameEntity*> childNodesCopy = childNodes;
404  for (GameEntity *child : childNodesCopy) {
405  // If still a child, update it
406  if (hasChild(child)) child->updateNode();
407  }
408 
409  // Remove dead
410  for (int i = 0; i < childNodes.size(); i++) {
411  if (!childNodes[i]->isAlive()) {
